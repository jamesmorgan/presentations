What is it?

Wikipedia

ECMAScript is the scripting language standardized by Ecma International in the ECMA-262 specification...
The language is widely used for client-side scripting on the web...several well-known implementations such as JavaScript, JScript and ActionScript..
Stackoverflow - Javascript is basically ECMAScript at it's core but builds upon it. Languages such as Actionscript, Javascript, JScript all use ECMAScript as its core.

New Features

  • Shorthand object literals
  • Inbuilt String templating
  • Default method params & var args
  • First class types of Map and Set not just Arrays and Functions
  • let - better variable scoping
  • const - readonly variable
  • => - Fat arrow syntax - lexical scoped functions
  • Generators & Iterators - yield keyword
  • Object deconstruction/swapping
  • Modules - (import/export) - classes with extensibility
  • Inbuilt Promises
  • Object.observe() - this will change lots of frameworks

Object literals & String templates


// Old school ECMAScript 5+ version
var home = "Germany";
var away = "Argentina";
var oldGame = {
    home: home,
    away: away
};
console.log(`Winners ${oldGame.home}`); // prints 'Winners Germany'

// New ECMAScript 6 way
var newGame = { home, away }; // fields are given implicit values
console.log(`Losers ${newGame.away}`); // prints 'Losers Argentina'

Default Params


var helloWorld = function helloWorld(name = "Jimbob"){
    return `${name} says hello!`;
};
console.log(helloWorld()); // prints 'Jimbob says hello!'
console.log(helloWorld("Billy")); // prints 'Billy says hello!'

Varargs

Also called rest params - an Array is populated with all trailing args (never null or undefined)


// Only applicable to use ... as last argument of method
function varargs(x, ...rest) {
    return rest.length;
}
console.log(varargs("a", 1,2,3,4,5,6)); // prints '6'
console.log(varargs("a")); // prints '0'

Sets & Maps

/** new API for accessing and creating maps - set,has,add,delete */
let m = new Map();
console.log(m.size);  // 0
m.set("a", 1);
m.has("a");           // true
m.get("a");           // 1
console.log(m.size);  // 1
m.delete("a");        // true
console.log(m.size);  // 0

Generators and yield keyword

An Iterator is an object that knows how to access items from a collection one at a time, while keeping track of its current position within that sequence.
Each call to the generator-iterator's next() method will execute the body of the function up to the next yield expression and return its result 

function* count() {
    yield 1;
    yield 2;
    yield 3;
}
var counter = count();
console.log(counter.next()); // {value: 1, done: false}
console.log(counter.next()); // {value: 2, done: false}
console.log(counter.next()); // {value: 3, done: false}
console.log(counter.next()); // {value: undefined, done: true}

Notice function*

Custom Iterators

Syntax has changed several times so expect changes in the future.

Generators enable lazy computation of sequences, with items calculated on-demand as they are needed. 

function fibonacci(){
    var fn1 = 1;
    var fn2 = 1;
    while (1){
        var current = fn2;
        fn2 = fn1;
        fn1 = fn1 + current;
        yield current;
    }
}
var sequence = fibonacci();
print(sequence.next()); // 1
print(sequence.next()); // 1
print(sequence.next()); // 2
print(sequence.next()); // 3
print(sequence.next()); // 5
print(sequence.next()); // 8
print(sequence.next()); // 13

Array Comprehension


var numbers = [1.5, 6, 8.5, 12];
var doubled = [for (i of numbers) i * 2];
console.log(doubled); // prints 3,12,17,24

// Can also be done as
var alternative = numbers.map(i => { return i * 2; });
console.log(alternative); // prints 3,12,17,24

// Fat arrow equivalent
var alternativeWay = numbers.map(function(i){
    return i * 2;
});
Similar to generators, a way of defining list creation. Also here is an example of fat arrow syntax, syntactical sugar only.

Array Reduction

Allows the result of each reduction to be use in the next one


// Make a reusable function
function adder(a, b) {
    return a + b;
}

var total = [0, 1, 2, 3].reduce(adder);
console.log(total); // prints '6'

function joiner(a, b) {
    return a.concat(b);
}
var flattened = [[0, 1], [2, 3], [4, 5]].reduce(joiner);
console.log(flattened); // flattened is [0, 1, 2, 3, 4, 5]

Array Spreading

Allows combine multiple arrays more easily.

Notice the ... notation


var a1 = [1];
var b1 = [2, 3, 4];
var c1 = [6, 7];
var d1 = [0, ...a1, ...b1, 5, ...c1];
console.log(d1); // prints [0, 1, 2, 3, 4, 5, 6, 7]

// ES5 - Current equivalent - messy
var d2 = [0].concat(a1).concat(b1).concat([5]).concat(c1);
console.log(d2); // prints [0, 1, 2, 3, 4, 5, 6, 7]

Create a simple logging class


export class Logger {
    constructor(name = 'logger', printDate = false){
        this.name = name;
        this.printDate = printDate;
    }
    debug(msg){
        this.printMsg('debug', msg);
    }
    info(msg){
        this.printMsg('info', msg);
    }
    warn(msg){
        this.printMsg('warn', msg);
    }
    error(msg){
        this.printMsg('error', msg);
    }
    printMsg(fnc, msg) {
        if(this.printDate){
            console[fnc](`[${this.date()}] [${this.name}] ${msg}`);
        } else {
            console[fnc](`[${this.name}] ${msg}`);
        }
    }
    date() {
        return moment().format('llll');
    }
}

Class Inheritance

Create a base class


class Animal {
    constructor(){
        this.type = "Animal";
        this.colour = "Brown";
        this.size = "Big";
    }
    // Method Definition Shorthand
    sound() {
        return "Sqeak";
    }
    move() {
        return "Running";
    }
    toString() {
        return `The ${this.size} ${this.colour} ${this.type} can ${this.sound()} and ${this.move()}`;
    }
}

Extended your classes


class Bird extends Animal {
    // Implicit constructor - an implicit super() call if not specified
    constructor(colour, size = "Tiny") {
        this.type = "Bird";
        this.colour = colour;
        this.size = size;
    }
    sound() { return "Cheep"; }
    move() { return "Flap" }
}

class Dog extends Animal {
    constructor(colour, size, type = "Dog") {
        this.type = type;
        this.colour = colour;
        this.size = size;
    }
    sound() { return "Woof"; }
}

class Labrador extends Dog {
    constructor(colour, size) {
        // Example of none implicit super() call
        super(colour, size, "Labrador")
    }
    move() { return "Sprint"; }
    sound() { return "Bark"; }
}

Using your classes


var c = new Bird("Red"); // Default arg on constructor
console.log(c.toString());

var d = new Dog("Black", "Medium");
console.log(d.toString());

var p = new Labrador("Brown", "Big");
console.log(p.toString());
